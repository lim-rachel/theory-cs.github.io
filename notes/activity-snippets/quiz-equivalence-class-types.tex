a set $U$:

\item $\forall x \in U ~([x]_E \in U)$
\item $\forall x \in U ~([x]_E \subseteq U)$
\item $\forall x \in U ~([x]_E \in \mathcal{P}(U))$
\item $\forall x \in U ~([x]_E \subseteq \mathcal{P}(U))$
\end{enumerate}
