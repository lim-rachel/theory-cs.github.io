Consider the following input-output definition tables with $X = \{ a,b,c\}$ and 
$Y = \{ 1, 2, 3, 4, 5\}$ and $Z = \{ 10, 20, 30 \}$

\begin{center}
\begin{multicols}{3}
\begin{tabular}{c}
Table 1\\
 \begin{tabular}{c|c}
Input & Output \\
\hline
$1$ & $10$ \\
$2$ & $20$ \\
$3$ & $30$ \\
\end{tabular}\end{tabular}
\columnbreak
\begin{tabular}{c}
Table 2\\ 
\begin{tabular}{c|c}
Input & Output \\
\hline
$a$ & $1$ \\
$b$ & $4$ \\
$c$ & $5$ \\
\end{tabular}\end{tabular}
\columnbreak
\begin{tabular}{c}
Table 3\\ 
\begin{tabular}{c|c}
Input & Output \\
\hline
$10$ & $a$ \\
$20$ & $b$ \\
$30$ & $a$ \\
\end{tabular}\end{tabular}
\end{multicols}
\end{center}

\begin{enumerate}
\item Select all and only the tables that each define a well-defined function 
whose domain and codomain 
is each $X$, $Y$, or $Z$.
\item Select all and only the tables that each define a well-defined function 
(with domain $X$ or $Y$ or $Z$ and 
with codomain $X$ or $Y$ or $Z$) and that is one-to-one.
\item Select all and only the tables that each define a well-defined function 
(with domain $X$ or $Y$ or $Z$ and 
with codomain $X$ or $Y$ or $Z$) and that is onto.
\end{enumerate}
